function updatePrintInfo(monthIdx) { const monthText = monthIdx === 'all' ? 'Todos los Meses' : MONTHS[monthIdx]; const typeText = filterMode === 'all' ? 'Todos los Registros' : (filterMode === 'income' ? 'Solo Ingresos' : 'Solo Egresos'); document.getElementById('print-filter-info').innerText = `${monthText} | Filtro: ${typeText}`; } function renderTable(dataToRender = records) { const tbody = document.getElementById('table-body'); tbody.innerHTML = ''; let totalIn = 0; let totalOut = 0; dataToRender.forEach((r, index) => { totalIn += r.income; totalOut += r.expense; const tr = document.createElement('tr'); tr.className = "hover:bg-slate-50 transition-colors"; tr.innerHTML = ` #${dataToRender.length - index} ${formatDate(r.date)} ${r.name} ${r.income > 0 ? '$' + r.income.toLocaleString() : '-'} ${r.expense > 0 ? '$' + r.expense.toLocaleString() : '-'} ${r.obs || '---'} `; tbody.appendChild(tr); }); updateStats(dataToRender.length, totalIn, totalOut); } function updateStats(count, income, expense) { const balance = income - expense; // Stats Panel document.getElementById('stat-count').innerText = count; document.getElementById('stat-income').innerText = '$' + income.toLocaleString(); document.getElementById('stat-expense').innerText = '$' + expense.toLocaleString(); document.getElementById('stat-balance').innerText = '$' + balance.toLocaleString(); document.getElementById('stat-balance').className = `text-2xl font-bold ${balance >= 0 ? 'text-slate-800' : 'text-rose-600'}`; // Print Totals document.getElementById('print-total-income').innerText = '$' + income.toLocaleString(); document.getElementById('print-total-expense').innerText = '$' + expense.toLocaleString(); document.getElementById('print-total-balance').innerText = '$' + balance.toLocaleString(); } function formatDate(dateStr) { const parts = dateStr.split('-'); return `${parts[2]}/${parts[1]}/${parts[0]}`; } // Event Listeners adicionales document.addEventListener('keydown', (e) => { if (e.key === 'Escape') toggleModal(false); });
